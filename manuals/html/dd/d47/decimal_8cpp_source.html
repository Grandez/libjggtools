14 Decimal::Decimal(const char* value) {
15 regex pat{"^[+-]?[0-9]*[\\.,]?[0-9]*$"};
16 bool match = regex_search(value, pat);
17 if (!match) throw new ToolsValueException(BAD_DECIMAL, value);
18 vector<char *> data = (sstring(value)).tokenize("[\\.,]");
19 if (data.size() > 2 || data.size() == 0) throw new ToolsValueException(BAD_DECIMAL, value);
20 number = atoll(data[0]);
21 scale = 0;
22 if (data.size() == 2) {
23 scale = (unsigned int) strlen(data[1]);
24 number *= pow(10,scale);
25 number += atol(data[1]);
26 }
27 // When using char *, those are allocated
28 for (int i = 0; i < data.size(); i++) free(data[i]);
29 }

168 char m[64];
169 int modulus = op->number1 % (10 ^ op->scale1);
