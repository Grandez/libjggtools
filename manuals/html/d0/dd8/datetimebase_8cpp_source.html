33 void DateTimeBase::fillTmDate(const char *str, int fmt) {
34 vector<int> res = sstring(str).tokenizeNumber((const char*)"[/-]");
35 time_base::dateorder d;
36 d = (fmt == -1) ? use_facet<time_get<char>>(locale()).date_order()
37 : static_cast<time_base::dateorder>(fmt);
38
39 switch (d) {
40 case time_base::dmy: mtm.tm_mday = res[0]; mtm.tm_mon = res[1] - 1; mtm.tm_year = res[2] - 1900; break;
41 case time_base::mdy: mtm.tm_mday = res[1]; mtm.tm_mon = res[0] - 1; mtm.tm_year = res[2] - 1900; break;
42 case time_base::ymd: mtm.tm_mday = res[2]; mtm.tm_mon = res[1] - 1; mtm.tm_year = res[2] - 1900; break;
43 case time_base::ydm: mtm.tm_mday = res[2]; mtm.tm_mon = res[2] - 1; mtm.tm_year = res[0] - 1900; break;
44 default: mtm.tm_mday = res[0]; mtm.tm_mon = res[1] - 1; mtm.tm_year = res[2] - 1900;
45 }

52 void DateTimeBase::validateDateValues (const char *str) {
53 int days[] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
54 if ( mtm.tm_year < 0) throw new ToolsValueException(BAD_DATE, str);
55 if ((mtm.tm_mon < 0) || (mtm.tm_mon > 11)) throw new ToolsValueException(BAD_DATE, str);
56 if ((mtm.tm_mday < 1) || (mtm.tm_mday > 31)) throw new ToolsValueException(BAD_DATE, str);
57 if ( mtm.tm_mon != 2) {
58 if (mtm.tm_mday > days[mtm.tm_mon]) throw new ToolsValueException(BAD_DATE, str);
59 else if (isLeap(mtm.tm_year) && mtm.tm_mday > 29) throw new ToolsValueException(BAD_DATE, str);
60 }
